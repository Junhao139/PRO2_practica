38

42  if (available > in_demand) {
43  srinfo.left_for_buy -= available - in_demand;
44  transacted = true;
45  }
46  }
47  if (city_obj.exist_in_inventory(ship.for_sell_product())) {
48  int in_demand, available;
49  city_obj.consult_product(ship.for_sell_product(), in_demand, available);
50  if (in_demand > available) {
51  srinfo.left_for_sell -= in_demand - available;
52  transacted = true;
53  }
54  }
55 
56  // if transacted too much, reset to zero
57  if (srinfo.left_for_buy < 0) srinfo.left_for_buy = 0;
58  if (srinfo.left_for_sell < 0) srinfo.left_for_sell = 0;
59 
60  // if no further trades are needed
61  if (srinfo.left_for_buy == 0 and srinfo.left_for_sell == 0) {
62  route.push_back(city.value());
63  return srinfo;
64  }
65 
66  list<string> left_subroute, right_subroute;
67 
68  SubrouteInfo left_accumulated = ship_find_optimum_route(left_subroute, city.left(), ship, srinfo);
69  SubrouteInfo right_accumulated = ship_find_optimum_route(right_subroute, city.right(), ship, srinfo);
70 
71  int sum_srinfo_this = srinfo.left_for_buy + srinfo.left_for_sell;
72  int sum_srinfo_left = left_accumulated.left_for_buy + left_accumulated.left_for_sell;
73  int sum_srinfo_right = right_accumulated.left_for_buy + right_accumulated.left_for_sell;
74 
75  // if no further trades are done
76  if (sum_srinfo_this == sum_srinfo_left and sum_srinfo_this == sum_srinfo_right) {
77  if (transacted) route.push_front(city.value());
78  return srinfo;
79  }
80  // if in both subroutes have done trades
81  else if (sum_srinfo_this != sum_srinfo_left and sum_srinfo_this != sum_srinfo_right) {
82  // compare in which route sold more
83  if (sum_srinfo_left < sum_srinfo_right) {
84  left_subroute.push_front(city.value());
85  route = left_subroute;
86  return left_accumulated;
87  } else if (sum_srinfo_right < sum_srinfo_left) {
88  right_subroute.push_front(city.value());
89  route = right_subroute;
90  return right_accumulated;
91  } else {

94  left_subroute.push_front(city.value());
95  route = left_subroute;

98  right_subroute.push_front(city.value());
99  route = right_subroute;
100  return right_accumulated;
101  } else {

106  }
107  }
108  }

123 void River::ship_transact(string const& city_name, Ship& ship, ProductData const& pddata)
124 {
125  auto city_it = this->inventory_database.find(city_name);
126 
127  int ship_wanted_product_count = ship.wanted_number();
128  int ship_for_sell_product_count = ship.for_sell_number();
129 
130  if (city_it->second.exist_in_inventory(ship.wanted_product())) {
131  int ship_wanted_in_demand, ship_wanted_available;
132  city_it->second.consult_product(ship.wanted_product(), ship_wanted_in_demand, ship_wanted_available);
133  // only transact if the city has more than it needs
134  if (ship_wanted_available > ship_wanted_in_demand) {
135  int available_for_purchase = ship_wanted_available - ship_wanted_in_demand;
136  if (ship.wanted_number() <= available_for_purchase) {
137  city_it->second.set_product_in_inventory(ship.wanted_product(), pddata, ship_wanted_in_demand, ship_wanted_available - ship.wanted_number());
138  // now the ship dows not need this product anymore
139  ship_wanted_product_count = 0;
140  } else {
141  // now the ship bought some, the inventory of the city reduced to exactly how many it needs
142  ship_wanted_product_count -= available_for_purchase;
143  city_it->second.set_product_in_inventory(ship.wanted_product(), pddata, ship_wanted_in_demand, ship_wanted_in_demand);
144  }
145  }
146  }
147 
148  if (city_it->second.exist_in_inventory(ship.for_sell_product())) {
149  int ship_for_sell_in_demand, ship_for_sell_available;
150  city_it->second.consult_product(ship.for_sell_product(), ship_for_sell_in_demand, ship_for_sell_available);
151  // only transact if the city needs more
152  if (ship_for_sell_in_demand > ship_for_sell_available) {
153  int available_for_sell = ship_for_sell_in_demand - ship_for_sell_available;
154  if (ship.for_sell_number() <= available_for_sell) {
155  city_it->second.set_product_in_inventory(ship.for_sell_product(), pddata, ship_for_sell_in_demand, ship_for_sell_available + ship.for_sell_number());
156  // now the ship sold all of what it has
157  ship_for_sell_product_count = 0;
158  } else {
159  // now the ship sold some, the inventory of the city augmented to exactly how many it needs
160  ship_for_sell_product_count -= available_for_sell;
161  city_it->second.set_product_in_inventory(ship.for_sell_product(), pddata, ship_for_sell_in_demand, ship_for_sell_in_demand);
162  }
163  }
164  }
165 
166  // update the ship's info after crossing with this city
167  ship.set_all(ship.wanted_product(), ship_wanted_product_count, ship.for_sell_product(), ship_for_sell_product_count);

225  // Before applying the transactions, store the original values for the step (4).
226  int ship_initial_wanted_count = ship.wanted_number();
227  int ship_initial_for_sell_count = ship.for_sell_number();
